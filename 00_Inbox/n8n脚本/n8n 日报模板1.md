```js
// ==========================================
// ⚙️ 配置区域
// ==========================================
const CONFIG = {
  typeField: 'n8n_source_type', // 类型字段名
  typeMap: {
    'togglTrack_integration': 'toggl',
    'todoist_integration': 'todoist'
    // 未来扩展: 'google_calendar_integration': 'calendar'
  },
  fields: {
    'toggl': {
      timeStart: 'start',
      timeStop: 'stop',
      title: 'description',
      project: 'n8n_isoft_pid',
      id: 'id',
      projectId: 'pid'
    },
    'todoist': {
      timeStart: 'posted_at',
      title: 'content',
      project: 'project_id',
      id: 'id', 
      projectId: 'task_id' 
    }
    // 未来扩展: 'calendar': { ... }
  }
};

// ==========================================
// 🚀 主逻辑区域
// ==========================================

const items = $input.all();
let md = `# 📅 每日工作日报 - ${new Date().toISOString().split('T')[0]}\n\n---\n\n### 📜 混合时间流\n\n`;

for (const item of items) {
  const json = item.json;
  const rawType = json[CONFIG.typeField];
  const type = CONFIG.typeMap[rawType];

  // 如果类型未定义，跳过
  if (!type || !CONFIG.fields[type]) continue;

  const fieldMap = CONFIG.fields[type];
  
  // --- 1. 基础信息准备 ---
  const startTimeVal = json[fieldMap.timeStart];
  let timeDisplay = '';
  let durationDisplay = '';
  let sourceData = {}; // 溯源数据

  if (startTimeVal) {
    const dateObj = new Date(startTimeVal);
    const timeStr = dateObj.toLocaleTimeString('zh-CN', {hour: '2-digit', minute:'2-digit'});
    
    // --- 2. 显式类型分支处理 ---
    
    // 处理 Toggl 类型
    if (type === 'toggl') {
      timeDisplay = `**${timeStr}**`;
      const stopTimeVal = json[fieldMap.timeStop];
      if (stopTimeVal) {
        const stopDate = new Date(stopTimeVal);
        const diffMinutes = Math.round((stopDate - dateObj) / 60000);
        durationDisplay = ` (${diffMinutes}m)`;
      } else {
        durationDisplay = `(进行中)`;
      }
      // 构建 Toggl 溯源数据
      sourceData = { id: json[fieldMap.id], pid: json[fieldMap.projectId] };
    }
    
    // 处理 Todoist 类型
    if (type === 'todoist') {
      timeDisplay = `**${timeStr}**`;
      // Todoist 没有时长
      durationDisplay = ''; 
      // 构建 Todoist 溯源数据
      sourceData = { comment_id: json[fieldMap.id], project_id: json[fieldMap.projectId] };
    }

    // 未来扩展: 处理 Calendar 类型
    // if (type === 'calendar') { ... }
  }

  // --- 3. 提取通用字段 ---
  const title = json[fieldMap.title] || '无标题';
  const projectName = json[fieldMap.project];

  // --- 4. 生成 HTML 溯源标签 ---
  const sourceHtml = `<details><summary>查看ID</summary>\`${JSON.stringify(sourceData)}\`</details>`;

  // --- 5. 拼接 Markdown ---
  const icon = type === 'toggl' ? '🟢' : '🔵';
  const typeLabel = type === 'toggl' ? '[Toggl]' : '[Todoist]';

  md += `- ${timeDisplay}${durationDisplay}\n`;
  md += `> **${icon} ${typeLabel} ${title}**\n`;
  
  if (projectName) {
      md += `- **项目**: ${projectName}\n`; 
  }
  
  md += `- **溯源**: ${sourceHtml}\n\n`;
}

// --- 6. 底部固定内容 ---
md += `---\n\n### 🧠 AI 智能总结\n\n> (此处等待 AI 生成...)\n\n---\n\n### 💾 原始数据归档\n\n<details>\n<summary>点击展开 JSON 原始数据</summary>\n\n\`\`\`json\n${JSON.stringify(items.map(i => i.json), null, 2)}\n\`\`\`\n</details>`;

return [{ json: { markdown_content: md } }];
```
